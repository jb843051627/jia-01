// AppointmentServiceImpl.insertAppointment()
public AjaxResult insertAppointment(Appointment appointment) {
    // 1. 参数校验
    if (appointment.getAppointmentTime() == null) {
        return AjaxResult.error("预约时间不能为空");
    }
    
    // 2. 根据服务类型计算结束时间和金额
    ServiceType serviceType = serviceTypeService.selectServiceTypeById(appointment.getServiceTypeId());
    Calendar calendar = Calendar.getInstance();
    calendar.setTime(appointment.getAppointmentTime());
    calendar.add(Calendar.MINUTE, serviceType.getDuration() != null ? serviceType.getDuration() : 120);
    appointment.setEndTime(calendar.getTime());
    appointment.setAmount(serviceType.getPrice());
    
    // 3. 时间冲突检查
    List<Appointment> conflicts = checkTimeConflict(appointment.getAppointmentTime(), appointment.getEndTime(), null);
    if (!conflicts.isEmpty()) {
        return AjaxResult.error("该时间段已有预约，请选择其他时间");
    }
    
    // 4. 生成预约单号，设置初始状态
    appointment.setAppointmentNo(generateAppointmentNo()); // 格式：AP + yyyyMMdd + 雪花ID后6位
    appointment.setStatus("0"); // 待确认
    
    // 5. 插入数据库
    int result = this.baseMapper.insert(appointment);
    
    // 6. 刷新客户档案
    if (result > 0) {
        customerProfileService.refreshOrCreateProfile(appointment.getCustomerName(), appointment.getCustomerPhone());
        return AjaxResult.success("预约成功");
    }
    return AjaxResult.error("预约失败");
}
```

public AjaxResult updateStatus(Long id, String status) {
    Appointment appointment = new Appointment();
    appointment.setId(id);
    appointment.setStatus(status);
    
    // 服务中状态：记录实际开始时间
    if (AppointmentStatus.SERVING.getCode().equals(status)) {
        appointment.setActualStartTime(new Date());
    } 
    // 已完成状态：记录实际结束时间
    else if (AppointmentStatus.COMPLETED.getCode().equals(status)) {
        appointment.setActualEndTime(new Date());
    }
    
    int result = this.baseMapper.updateById(appointment);
    
    if (result > 0) {
        // 进入服务中状态时，创建服务流程节点
        if (AppointmentStatus.SERVING.getCode().equals(status)) {
            serviceFlowService.createFlowNodes(id);
        }
        // 刷新客户档案
        Appointment fullAppointment = this.baseMapper.selectById(id);
        if (fullAppointment != null) {
            customerProfileService.refreshOrCreateProfile(
                fullAppointment.getCustomerName(), 
                fullAppointment.getCustomerPhone()
            );
        }
        return AjaxResult.success("状态更新成功");
    }
    return AjaxResult.error("状态更新失败");
}
```

// ServiceFlowServiceImpl.createFlowNodes()
private static final String[][] FLOW_NODES = {
    {"reception", "接待"},
    {"farewell", "告别"},
    {"cremation", "火化"},
    {"ashes", "骨灰安置"}
};

@Transactional(rollbackFor = Exception.class)
public AjaxResult createFlowNodes(Long appointmentId) {
    for (int i = 0; i < FLOW_NODES.length; i++) {
        ServiceFlow node = new ServiceFlow();
        node.setAppointmentId(appointmentId);
        node.setNodeCode(FLOW_NODES[i][0]);
        node.setNodeName(FLOW_NODES[i][1]);
        node.setSortOrder(i + 1);
        // 第一个节点默认激活（进行中）
        node.setStatus(i == 0 ? STATUS_ACTIVE : STATUS_PENDING);
        this.baseMapper.insert(node);
    }
    return AjaxResult.success("流程节点创建成功");
}
```

@Transactional(rollbackFor = Exception.class)
public AjaxResult completeNode(Long id, String remark) {
    ServiceFlow node = this.baseMapper.selectById(id);
    
    // 状态校验
    if (STATUS_COMPLETED.equals(node.getStatus())) {
        return AjaxResult.error("节点已完成");
    }
    if (STATUS_PENDING.equals(node.getStatus())) {
        return AjaxResult.error("当前节点未激活，无法完成");
    }
    
    // 标记当前节点为已完成
    node.setStatus(STATUS_COMPLETED);
    node.setCompleteTime(new Date());
    node.setRemark(remark);
    this.baseMapper.updateById(node);
    
    // 激活下一节点
    List<ServiceFlow> allNodes = serviceFlowMapper.selectServiceFlowByAppointmentId(node.getAppointmentId());
    for (ServiceFlow n : allNodes) {
        if (n.getSortOrder() == node.getSortOrder() + 1 && STATUS_PENDING.equals(n.getStatus())) {
            n.setStatus(STATUS_ACTIVE);
            this.baseMapper.updateById(n);
            break;
        }
    }
    return AjaxResult.success("节点已完成");
}
```

**修改的表**：
- `t_service_flow`：当前节点 status="2"（已完成），complete_time=now

// [AdminController.java](file:///d:/Develop/Workspace-trae/jia-01/src/main/java/com/fc/v2/controller/AdminController.java#L106-L174)
@PostMapping("/login")
@ResponseBody
public AjaxResult login(TSysUser user, String captcha, boolean rememberMe, HttpServletRequest request) {
    // 1. 验证码校验
    String verCode = (String) request.getSession().getAttribute("captcha");
    if (captcha == null || !captcha.toLowerCase().equals(verCode.trim().toLowerCase())) {
        return AjaxResult.error(500, "验证码不正确!");
    }
    CaptchaUtil.clear(request);
    
    // 2. 获取 Subject，创建 Token
    Subject currentUser = SecurityUtils.getSubject();
    UsernamePasswordToken token = new UsernamePasswordToken(user.getUsername(), user.getPassword());
    
    if (rememberMe) {
        token.setRememberMe(true);
    }
    
    // 3. 执行登录
    currentUser.login(token);
    
    if (StringUtils.isNotNull(ShiroUtils.getUser())) {
        return AjaxResult.success();
    }
    return AjaxResult.error(500, "未知账户");
}
```

protected AuthenticationInfo doGetAuthenticationInfo(AuthenticationToken token) throws AuthenticationException {
    if (StringUtils.isNull(token.getPrincipal())) {
        return null;
    }
    
    String username = (String) token.getPrincipal();
    String password = new String((char[]) token.getCredentials());

    userInfo.setUsername(username);
    List<TSysUser> list = userService.selectTSysUserList(userInfo);
    
    if (CollUtil.isEmpty(list)) {
        return null;

    return new SimpleAuthenticationInfo(userInfo, userInfo.getPassword(), getName());
}

### 2.4 授权流程（权限校验）

#### 2.4.1 调用链路

        ↓

                            ↓
                            查询用户角色 → 查询角色权限
                            ↓
                            返回 SimpleAuthorizationInfo（包含角色和权限）
                                ↓

```

#### 2.4.2 关键代码

**授权逻辑 - MyShiroRealm**：
```java
// [MyShiroRealm.java](file:///d:/Develop/Workspace-trae/jia-01/src/main/java/com/fc/v2/shiro/service/MyShiroRealm.java#L91-L115)
@Override
protected AuthorizationInfo doGetAuthorizationInfo(PrincipalCollection principals) {
    if (StringUtils.isNull(principals)) {
        throw new AuthorizationException("principals should not be null");
    }
    
    SimpleAuthorizationInfo authorizationInfo = new SimpleAuthorizationInfo();
    TSysUser userinfo = (TSysUser) principals.getPrimaryPrincipal();
    Long uid = userinfo.getId();
    
    // 查询用户角色
    List<TSysRole> tsysRoles = roleService.queryUserRole(uid);
    tsysRoles.forEach(userrole -> {
        // 添加角色
        authorizationInfo.addRole(userrole.getName());
        
        // 查询角色权限
        List<TSysPermission> permissions = permissionService.queryRoleId(userrole.getId());
        permissions.forEach(permission -> {
            // 添加权限字符串
            if (StringUtils.isNotEmpty(permission.getPerms())) {
                authorizationInfo.addStringPermission(permission.getPerms());
            }
        });
    });
    
    return authorizationInfo;
}
```

## 3. MyBatis-Plus 自动填充机制

### 3.1 架构概述

MyBatis-Plus 自动填充功能用于在插入或更新数据时，自动为指定字段赋值（如创建时间、更新时间、创建人、更新人等），避免重复代码。

### 3.2 核心组件

| 组件 | 路径 | 职责 |
|------|------|------|
| MetaObjectHandler | [MybatisPlusConfig.java](file:///d:/Develop/Workspace-trae/jia-01/src/main/java/com/fc/v2/common/conf/MybatisPlusConfig.java#L65-L84) | 填充策略实现 |
| BaseEntity | [BaseEntity.java](file:///d:/Develop/Workspace-trae/jia-01/src/main/java/com/fc/v2/common/base/BaseEntity.java) | 基础实体类，定义自动填充字段 |
| @TableField 注解 | - | 标记需要自动填充的字段 |

### 3.3 调用链路

```
执行 INSERT / UPDATE 操作
        ↓
MyBatis-Plus 拦截器（MetaObjectHandler）
        ↓
根据 @TableField(fill = ...) 判断是否需要填充
        ↓
┌──────────────────────────────────┐
│  INSERT 操作 → insertFill()      │
│  UPDATE 操作 → updateFill()      │
└──────────────────────────────────┘
        ↓
        为字段赋值后，执行 SQL
```

### 3.4 关键代码

#### 3.4.1 实体类字段标记

```java
// [BaseEntity.java](file:///d:/Develop/Workspace-trae/jia-01/src/main/java/com/fc/v2/common/base/BaseEntity.java#L19-L39)
public class BaseEntity implements Serializable {
    
    /** 创建者 */
    @TableField(value = "create_by", fill = FieldFill.INSERT)  // 插入时填充
    private String createBy;
    
    /** 创建时间 */
    @TableField(value = "create_time", fill = FieldFill.INSERT)  // 插入时填充
    @JsonFormat(pattern = "yyyy-MM-dd HH:mm:ss", timezone="GMT+8")
    private Date createTime;
    
    /** 更新者 */
    @TableField(value = "update_by", fill = FieldFill.UPDATE)  // 更新时填充
    private String updateBy;
    
    /** 更新时间 */
    @TableField(value = "update_time", fill = FieldFill.UPDATE)  // 更新时填充
    @JsonFormat(pattern = "yyyy-MM-dd HH:mm:ss", timezone="GMT+8")
    private Date updateTime;
    
    /** 备注 */
    @TableField(value = "remark")  // 不自动填充
    private String remark;
}
```

**FieldFill 枚举值说明**：
- `FieldFill.DEFAULT`：默认不处理
- `FieldFill.INSERT`：插入时填充
- `FieldFill.UPDATE`：更新时填充
- `FieldFill.INSERT_UPDATE`：插入和更新时都填充

#### 3.4.2 填充策略实现

```java
// [MybatisPlusConfig.java](file:///d:/Develop/Workspace-trae/jia-01/src/main/java/com/fc/v2/common/conf/MybatisPlusConfig.java#L62-L85)
@Bean
@ConfigurationProperties("mybatis-plus.global-config")
public GlobalConfig globalConfig() {
    return new GlobalConfig().setMetaObjectHandler(new MetaObjectHandler() {
        
        /**
         * 插入时自动填充
         */
        @Override
        public void insertFill(MetaObject metaObject) {
            // 填充 createBy：当前登录用户名
            this.fillStrategy(metaObject, "createBy", ShiroUtils.getLoginName());
            // 填充 createTime：当前时间
            this.strictInsertFill(metaObject, "createTime", Date.class, new Date());
        }
        
        /**
         * 更新时自动填充
         */
        @Override
        public void updateFill(MetaObject metaObject) {
            // 填充 updateBy：当前登录用户名
            this.fillStrategy(metaObject, "updateBy", ShiroUtils.getLoginName());
            // 填充 updateTime：当前时间
            this.strictUpdateFill(metaObject, "updateTime", Date.class, new Date());
        }
    });
}
```

public static String getLoginName() {
    try {
        Subject subject = SecurityUtils.getSubject();
        if (subject != null && subject.getPrincipal() != null) {
            TSysUser user = (TSysUser) subject.getPrincipal();
            return user.getUsername();

所有继承 `BaseEntity` 的实体类都会自动获得自动填充功能：

```java

### 3.7 注意事项

1. **字段必须存在**：MetaObject 中必须存在对应的字段和 setter 方法
2. **类型匹配**：填充值的类型必须与字段类型一致
3. **未登录处理**：`ShiroUtils.getLoginName()` 在未登录时返回空字符串，避免空指针
4. **fillStrategy vs strictInsertFill**：
   - `fillStrategy`：有值时不覆盖，空值时填充
   - `strictInsertFill` / `strictUpdateFill`：严格模式，强制填充

### 4.1 技术栈说明

### 4.3 整体守卫流程

```
用户访问 URL
    │
    ├─→ 后端 Shiro 过滤器拦截（第一层守卫）
    │    ├─→ 白名单（anon）：放行（/static/**, /admin/login, /captcha/**等）
    │    ├─→ 需认证（user）：检查是否登录

    │    │    └─→ 未登录 → 重定向到 /admin/login
    │    └─→ 需权限（perms）：检查权限
    │         ├─→ 有权限 → 放行
    │         └─→ 无权限 → 重定向到 /error/403
    │
    └─→ 前端页面加载（第二层守卫）
         ├─→ 登录页面：检查是否已登录，已登录则跳转到首页

### 4.8 守卫机制总结

| 层级 | 守卫点 | 实现方式 | 作用 |
|------|--------|----------|------|
| 第一层 | URL 拦截 | Shiro 过滤器链 | 全局访问控制，未登录跳转登录页 |
| 第二层 | 登录页面 | 后端检查登录状态 | 已登录用户访问登录页时自动跳转首页 |
| 第三层 | 方法权限 | @RequiresPermissions 注解 | 细粒度接口权限控制 |
| 第四层 | 菜单权限 | 后端返回用户菜单 | 前端只展示用户有权限的菜单 |
| 第五层 | Session 过期 | 前端 JS 检查 | 处理被踢下线、Session过期等场景 |
